Variables Part 2:

Variables are a way for your program to remember. You can think of them as place holders.

References

Objects are stored in memory (RAM). Each object is stored in a separate part of memory 


  var arr1 = ['a'];
  var arr2 = ['a'];
  console.log(arr1 === arr2); // -> false

Passing the objects into functions doesn't create a new object. It just uses the current one. This is called passing by reference. 


  function copyPlayer(player, name) {
      player.name = name;
      return player;
  }
  let player1 = {
      name: 'Mario',
      hitPoints: 30
  };
  let player2 = copyPlayer(player, 'Luigi');
  console.log(player1); // -> { name: 'Luigi', hitPoints: 30 }
  console.log(player2); // -> { name: 'Luigi', hitPoints: 30 }

Most of the time this is what you want. But if you don't want to pass by reference, you need to clone the object yourself. 


  function copyPlayer(player, name) {
    let newPlayer = {}; // create a new object
    newPlayer.name = name;
    return newPlayer;
  }
  let player1 = {
    name: 'Mario',
    hitPoints: 30
  };
  let player2 = copyPlayer(player, 'Luigi');
  console.log(player1); // -> { name: 'Mario', hitPoints: 30 }
  console.log(player2); // -> { name: 'Luigi', hitPoints: 10 }

If objects are too big to clone by hand then convert them to a JSON String and back to an Object and now you have a new Ojbect with all the properties of the original 


  function copyPlayer(player, name) {
    let newPlayer = JSON.parse(JSON.stringify(player));
    newPlayer.name = name;
    return newPlayer;
  }
  let player1 = {
    name: 'Mario',
    hitPoints: 30
  };
  let player2 = copyPlayer(player, 'Luigi', 10);
  console.log(player1); // -> { name: 'Mario', hp: 30 }
  console.log(player2); // -> { name: 'Luigi', hp: 10 }

Variable functions

Functions can be stored in variables. This is a very useful tool and a big chunk of how functional programming works. 


function add(a,b){
  return a+b;
}
let addMe = add;
console.log(addMe(3,4)) // 7

Looping through Arrays

Iterating (looping) through an array of numbers using a for loop 


  array = [ 1, 2, 3, 4, 5, 6 ]; 
  for (let i = 0; i < array.length; i++) { 
      console.log(array[i]); 
  }

Looping through Objects

Iterating (looping) through all the object properties


  let object = {
    variable_1:'value 1',
    variable_2:'value 2',
    variable_3:'value 3',
  }
  for (var property in object) {
    // to avoid issues you need to check if the property is in the Object
    if (object.hasOwnProperty(property)) {
      console.log(property); // value 1, value 2, value 3
    }
  }

Looping through Dictionaries

Objects and dictionaries are the same thing 


  const fruits = {
    apple: 28,
    orange: 17,
    pear: 54,
  }
  
  const keys = Object.keys(fruits);
  console.log(keys); // [apple, orange, pear] 
  const values = Object.values(fruits);
  console.log(values); // [28, 17, 54] 
  
  const keys = Object.keys(fruits)
  for (const key of keys) {
    console.log(key); // apple, orange, pear
  }
